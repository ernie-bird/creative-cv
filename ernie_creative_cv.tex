\resheading{Research in Composition}

\ressubheadingitems{Elements of Applied Character-Driven Composition}{A Guide for Emerging Composers}{Author}{2023 - Present}
{\resitem{Emphasizing the importance of empathy, honesty, and self-awareness in composition, encouraging composers to relate to their characters and avoid judgment or bias in their music}
	\resitem{Dynamic Musical Objects (DMOs) are introduced as musical abstractions representing characters or elements of a story. These musical sequences can evolve, attach to characters, and exist semi-independently from the composer's inner "I".}
	\resitem{Providing composers with tools to deepen their understanding of character-driven composition through thought experiments, music theory exercises, and philosophical insight}
}
